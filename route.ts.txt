function getTimeSync() {
  const now = new Date();
  const currentYear = now.getFullYear();
  const currentMonth = now.getMonth() + 1;
  return `[시간 동기화] 현재 날짜는 ${currentYear}년 ${currentMonth}월입니다. 절대 ${currentYear - 2}년이나 ${currentYear - 1}년 등 과거 연도를 언급하지 마세요.`;
}
function getKstDateKey() {
  const now = new Date();
  const formatter = new Intl.DateTimeFormat("en-CA", {
    timeZone: "Asia/Seoul",
    year: "numeric",
    month: "2-digit",
    day: "2-digit",
  });
  return formatter.format(now);
}
function hashString(input: string) {
  let hash = 0;
  for (let i = 0; i < input.length; i += 1) {
    hash = (hash * 31 + input.charCodeAt(i)) >>> 0;
  }
  return hash;
}

function getDeterministicScore(baseScore: number, seed: string) {
  const hash = hashString(seed);
  const offset = (hash % 19) - 9; // -9 ~ +9
  return Math.max(50, Math.min(100, baseScore + offset));
}

function pickDeterministic<T>(items: T[], seed: string) {
  const hash = hashString(seed);
  return items[hash % items.length];
}
